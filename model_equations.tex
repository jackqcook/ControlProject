\noindent
This document defines the 9-state quadrotor model used in the project (no position states).

%--------------------------------------------------------------------
\section*{State, input, and thrust}
%--------------------------------------------------------------------

\textbf{State vector} (inertial linear velocity, ZYX Euler angles, body angular rates):
\begin{equation}
  \bm{x} =
  \begin{bmatrix}
    v_x \\ v_y \\ v_z \\ \phi \\ \theta \\ \psi \\ p \\ q \\ r
  \end{bmatrix}
  \in \mathbb{R}^9,
  \qquad
  \bm{v} = \begin{bmatrix} v_x \\ v_y \\ v_z \end{bmatrix},\;
  \bm{\eta} = \begin{bmatrix} \phi \\ \theta \\ \psi \end{bmatrix},\;
  \bm{\omega} = \begin{bmatrix} p \\ q \\ r \end{bmatrix}.
\end{equation}

\textbf{Control input} (per-rotor thrusts):
\begin{equation}
  \bm{u} =
  \begin{bmatrix} u_1 \\ u_2 \\ u_3 \\ u_4 \end{bmatrix}
  \in \mathbb{R}^4,
  \qquad
  T = u_1 + u_2 + u_3 + u_4 \quad \text{(total thrust).}
\end{equation}

\textbf{Parameters:} mass $m$, gravity $g$, arm length $l$, yaw torque coefficient $c_\tau$,
inertia $\bm{J} = \mathrm{diag}(J_x,J_y,J_z)$, linear drag $\bm{D}_v = \mathrm{diag}(d_x,d_y,d_z)$,
and $\bm{e}_3 = [0,\,0,\,1]^\top$.

%--------------------------------------------------------------------
\section*{Nonlinear dynamics $\dot{\bm{x}} = \bm{f}(\bm{x},\bm{u})$}
%--------------------------------------------------------------------

The dynamics split into translation, attitude kinematics, and rotation:
\begin{equation}
  \dot{\bm{x}} =
  \begin{bmatrix}
    \dot{\bm{v}} \\ \dot{\bm{\eta}} \\ \dot{\bm{\omega}}
  \end{bmatrix}
  =
  \begin{bmatrix}
    \dfrac{T}{m}\,\bm{R}(\phi,\theta,\psi)\,\bm{e}_3
    - g\,\bm{e}_3
    - \dfrac{1}{m}\,\bm{D}_v\,\bm{v}
    \\[0.6em]
    \bm{W}(\phi,\theta)\,\bm{\omega}
    \\[0.3em]
    \bm{J}^{-1}\bigl( \bm{\tau}(\bm{u}) - \bm{\omega} \times (\bm{J}\bm{\omega}) \bigr)
  \end{bmatrix}.
\end{equation}

%--------------------------------------------------------------------
\section*{Rotation matrix (ZYX Euler)}
%--------------------------------------------------------------------

\begin{equation}
  \bm{R}(\phi,\theta,\psi)
  = \bm{R}_z(\psi)\,\bm{R}_y(\theta)\,\bm{R}_x(\phi).
\end{equation}

The thrust direction in the inertial frame is $\bm{R}\,\bm{e}_3$:
\begin{equation}
  \bm{R}(\phi,\theta,\psi)\,\bm{e}_3
  =
  \begin{bmatrix}
    \cos\psi\,\sin\theta\,\cos\phi + \sin\psi\,\sin\phi \\
    \sin\psi\,\sin\theta\,\cos\phi - \cos\psi\,\sin\phi \\
    \cos\theta\,\cos\phi
  \end{bmatrix}.
\end{equation}

Component-wise translational acceleration:
\begin{align}
  \dot{v}_x &= \frac{T}{m}\bigl(\cos\psi\,\sin\theta\,\cos\phi + \sin\psi\,\sin\phi\bigr)
             - \frac{d_x}{m}\,v_x, \\
  \dot{v}_y &= \frac{T}{m}\bigl(\sin\psi\,\sin\theta\,\cos\phi - \cos\psi\,\sin\phi\bigr)
             - \frac{d_y}{m}\,v_y, \\
  \dot{v}_z &= \frac{T}{m}\,\cos\theta\,\cos\phi - g - \frac{d_z}{m}\,v_z.
\end{align}

%--------------------------------------------------------------------
\section*{Euler-angle kinematics}
%--------------------------------------------------------------------

With $\dot{\bm{\eta}} = \bm{W}(\phi,\theta)\,\bm{\omega}$,
\begin{equation}
  \bm{W}(\phi,\theta)
  =
  \begin{bmatrix}
    1 & \sin\phi\,\tan\theta & \cos\phi\,\tan\theta \\
    0 & \cos\phi & -\sin\phi \\
    0 & \dfrac{\sin\phi}{\cos\theta} & \dfrac{\cos\phi}{\cos\theta}
  \end{bmatrix}.
\end{equation}

Expanded (same as $\bm{W}\bm{\omega}$):
\begin{align}
  \dot{\phi}   &= p + q\,\sin\phi\,\tan\theta + r\,\cos\phi\,\tan\theta, \\
  \dot{\theta} &= q\,\cos\phi - r\,\sin\phi, \\
  \dot{\psi}   &= q\,\frac{\sin\phi}{\cos\theta} + r\,\frac{\cos\phi}{\cos\theta}.
\end{align}

%--------------------------------------------------------------------
\section*{Rotor torque map and rotational dynamics}
%--------------------------------------------------------------------

\textbf{Torque} generated by differential thrust (standard ``X'' or ``$+$'' frame mapping as in the notebook):
\begin{equation}
  \bm{\tau}(\bm{u})
  =
  \begin{bmatrix}
    l\,(u_2 - u_4) \\
    l\,(u_3 - u_1) \\
    c_\tau\,(u_1 - u_2 + u_3 - u_4)
  \end{bmatrix}.
\end{equation}

\textbf{Gyroscopic coupling:}
\begin{equation}
  \bm{\omega} \times (\bm{J}\bm{\omega})
  =
  \begin{bmatrix}
    (J_z - J_y)\,q\,r \\
    (J_x - J_z)\,p\,r \\
    (J_y - J_x)\,p\,q
  \end{bmatrix}.
\end{equation}

With diagonal $\bm{J}$, the body angular accelerations are
\begin{align}
  \dot{p} &= \frac{1}{J_x}\Bigl( l(u_2 - u_4) - (J_z - J_y)\,q\,r \Bigr), \\
  \dot{q} &= \frac{1}{J_y}\Bigl( l(u_3 - u_1) - (J_x - J_z)\,p\,r \Bigr), \\
  \dot{r} &= \frac{1}{J_z}\Bigl( c_\tau(u_1 - u_2 + u_3 - u_4) - (J_y - J_x)\,p\,q \Bigr).
\end{align}

%--------------------------------------------------------------------
\section*{Remarks}
%--------------------------------------------------------------------

\begin{itemize}
  \item Position $(X,Y,Z)$ is not in $\bm{x}$; this model describes velocity, attitude, and body rates only.
  \item At hover trim, $\bm{x}_0 = \bm{0}$ and $u_{i,0} = mg/4$ so $T_0 = mg$ and $\bm{f}(\bm{x}_0,\bm{u}_0)=\bm{0}$.
  \item Linearization for LQR/LQG uses
  $\bm{A} = \partial \bm{f}/\partial \bm{x}|_{\bm{x}_0,\bm{u}_0}$,
  $\bm{B} = \partial \bm{f}/\partial \bm{u}|_{\bm{x}_0,\bm{u}_0}$.
\end{itemize}
